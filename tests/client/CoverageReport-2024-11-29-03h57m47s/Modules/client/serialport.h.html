#include <string>

class ISerialPort {
public:
    virtual ~ISerialPort() = default;
    virtual bool connect(const std::string& portName, int baudRate) = 0;
    virtual void disconnect() = 0;
    virtual std::string sendMessage(const std::string& message) = 0;
    virtual bool SetCommState(HANDLE hSerial, DCB* dcb) = 0;
    virtual bool SetCommTimeouts(HANDLE hSerial, COMMTIMEOUTS* timeouts) = 0;
};

class SerialCommunication : public ISerialPort {
public:
    bool connect(const std::string& portName, int baudRate) override;
    void disconnect() override;
    std::string sendMessage(const std::string& message) override;
    bool SetCommState(HANDLE hSerial, DCB* dcb) override;
    bool SetCommTimeouts(HANDLE hSerial, COMMTIMEOUTS* timeouts) override;
    void drawBoard(const std::string& boardState);

private:
    HANDLE hSerial = INVALID_HANDLE_VALUE;
};
